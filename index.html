<script>

    const apiKey = "AIzaSyDIY6SPxJRHryZ-9FWTBE_te-bKXEn2TfM";

    let currentLang = 'en';


    const translations = {

        en: { welcome: "Welcome to Milan! ✨ I am Milo, Alessandra's virtual assistant. How can I help you today?", placeholder: "Ask Milo...", flag: "🇬🇧" },

        it: { welcome: "Benvenuto a Milano! ✨ Io sono Milo, l'assistente virtuale di Alessandra. Come posso aiutarti oggi?", placeholder: "Chiedi a Milo...", flag: "🇮🇹" },

        es: { welcome: "¡Bienvenido a Milán! ✨ Soy Milo, el asistente virtual de Alessandra. ¿Cómo posso ayudarte?", placeholder: "Pregunta a Milo...", flag: "🇪🇸" },

        fr: { welcome: "Bienvenue à Milan ! ✨ Je suis Milo, l'assistant virtuel d'Alessandra. Comment puis-je vous aider ?", placeholder: "Demandez à Milo...", flag: "🇫🇷" }

    };


    function toggleLang() { document.getElementById('lang-menu').classList.toggle('hidden'); }


    function setLang(lang) {

        currentLang = lang;

        document.getElementById('active-flag').innerText = translations[lang].flag;

        document.getElementById('welcome-txt').innerText = translations[lang].welcome;

        document.getElementById('user-input').placeholder = translations[lang].placeholder;

        toggleLang();

    }


    function addMessage(text, type) {

        const chat = document.getElementById('chat-container');

        const div = document.createElement('div');

        div.className = type === 'bot' ? 'bot-bubble' : 'user-bubble';

        div.innerHTML = `<p class="text-sm">${text.replace(/\n/g, '<br>')}</p>`;

        chat.appendChild(div);

        chat.scrollTop = chat.scrollHeight;

        return div;

    }


    async function handleSend(e) {

        e.preventDefault();

        const input = document.getElementById('user-input');

        const msg = input.value.trim();

        if(!msg) return;


        addMessage(msg, 'user');

        input.value = '';


        const loader = addMessage('<div class="flex gap-1"><div class="typing-dot"></div><div class="typing-dot"></div><div class="typing-dot"></div></div>', 'bot');


        const systemPrompt = `You are Milo, the virtual concierge for the "Bright 21" apartment in Milan, via Sammartini.

        Owner: Alessandra.

        WiFi: Credentials on the laminated sheet in the living room (use the QR code).

        Location: Near Central Station.

        Language: Respond only in ${currentLang}.

        Be polite, concise, and helpful. Use Google Search for local tips.`;


        try {

            const resp = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`, {

                method: 'POST',

                headers: { 'Content-Type': 'application/json' },

                body: JSON.stringify({

                    contents: [{ parts: [{ text: msg }] }],

                    systemInstruction: { parts: [{ text: systemPrompt }] },

                    tools: [{ "google_search": {} }]

                })

            });

            const data = await resp.json();

            loader.remove();

            const reply = data.candidates[0].content.parts[0].text;

            addMessage(reply, 'bot');

        } catch (err) {

            loader.remove();

            addMessage("I'm sorry, I have a connection problem. Please try again later.", 'bot');

        }

    }


    // Chiudi menu lingue se clicchi fuori

    window.onclick = function(event) {

        if (!event.target.closest('.group')) {

            document.getElementById('lang-menu').classList.add('hidden');

        }

    }

</script>
